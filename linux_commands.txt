set fedora mirror repository:
sudo sed -i '/^metalink=/s/^/#/' /etc/yum.repos.d/*.repo
sudo sed -i '/^#metalink=/i baseurl=https://mirror.mobinhost.com/fedora/linux/releases/\$releasever/Everything/\$basearch/os/' /etc/yum.repos.d/*.repo

setup ssh-server
sudo dnf install openssh-server
sudo systemctl start sshd
sudo systemctl enable sshd
sudo firewall-cmd --add-service=ssh --permanent

firewall:
sudo systemctl start firewalld
sudo systemctl enable firewalld
sudo firewall-cmd --list-all
sudo firewall-cmd --add-service=ssh --permanent
sudo firewall-cmd --remove-port=1025-65535/tcp --permanent
sudo firewall-cmd --remove-port=1025-65535/udp --permanent
sudo firewall-cmd --add-port=51820/udp --permanent
sudo firewall-cmd --reload

setup ghostty:
sudo dnf copr enable scottames/ghostty
sudo dnf install ghostty

    sudo rpm --import https://packages.microsoft.com/keys/microsoft.asc &&

sudo dnf check-update &&
sudo dnf install code # or code-insiders
